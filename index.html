function enviarUbicacionWhatsApp() { if (navigator.geolocation) { navigator.geolocation.getCurrentPosition( function(pos) { const lat = pos.coords.latitude.toFixed(6); const lon = pos.coords.longitude.toFixed(6); const mensaje = encodeURIComponent(`Hola, esta es mi ubicación actual:\nhttps://www.google.com/maps?q=${lat},${lon}`); // Cambia el número por el tuyo, sin "+" const numeroWhatsApp = '593968963177'; const url = `https://wa.me/${numeroWhatsApp}?text=${mensaje}`; window.open(url, '_blank'); }, function(error) { alert('Error al obtener la ubicación: ' + error.message); } ); } else { alert('Tu navegador no soporta geolocalización.'); } }
